
swap: function void (out x: integer, out y: integer) {
    temp: integer = x;
    x = y;
    y = temp;
    return;
}
bubbleSort: function void (out a: array[1] of integer, size: integer) {
    i: integer;
    for (i = 1, i < size-1, 1) {
        for (j = i+1, j < size, 1) {
            if (a[i] < a[j]) swap(a[i], a[j]);
        } 
    }
    return;
}
main: function void(){
    n: integer = readInteger();
    a: array[10] of integer = {1,2,3,4,5,6,7,8,9};
    bubbleSort(a, n);
}
